983  bool radial_sort(index_t bndl, RadialSort& RS) {
984  geo_debug_assert(bndl < nb());
985  if(nb_halfedges(bndl) <= 2) {
986  return true;
987  }
988  auto b = H_.begin() + std::ptrdiff_t(bndl_start_[bndl]);
989  auto e = H_.begin() + std::ptrdiff_t(bndl_start_[bndl+1]);
990  RS.init(*b);
991  std::sort(
992  b, e,
993  [&](index_t h1, index_t h2)->bool {
994  return RS(h1,h2);
995  }
996  );
997  bool OK = !RS.degenerate();
998  bndl_is_sorted_[bndl] = OK;
999  return OK;
1000  }
